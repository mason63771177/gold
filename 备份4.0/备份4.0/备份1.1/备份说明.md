## 技术实现

### 弹窗队列系统
```javascript
// 弹窗队列管理
let dialogQueue = [];
let isDialogShowing = false;

// 队列化显示弹窗
function showDialog(message, type, callback) {
    if (isDialogShowing) {
        dialogQueue.push({message, type, callback});
        return;
    }
    // 显示弹窗逻辑...
}

// 处理队列中的下一个弹窗
function closeDialog() {
    // 关闭当前弹窗后，延迟处理队列
    setTimeout(() => {
        if (dialogQueue.length > 0) {
            const next = dialogQueue.shift();
            showDialog(next.message, next.type, next.callback);
        }
    }, 500);
}
```
